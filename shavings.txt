export default function BookDetails() {
  const [bookDetails, setBookDetails] = useState({});
  const { id } = useParams();
  const navigate = useNavigate();

      setBookDetails(await getSingleBook(id));

        <button

            navigate("/");

        >
          Close Description
        </button>
      </div>
    </>
  );
}
